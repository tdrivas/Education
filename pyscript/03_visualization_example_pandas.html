Visualization of Natural gas prices with Twitter sentiment scores using Pandas and Matplotlib

import pandas as pd import matplotlib.pyplot as plt from pyodide.http import open_url df = pd.read_csv(open_url("https://raw.githubusercontent.com/tdrivas/Education/main/datasets/data.csv")) df.columns x = df["Price"] y = df["Sentiment"] fig, ax = plt.subplots() ax.plot(x, y) fig
